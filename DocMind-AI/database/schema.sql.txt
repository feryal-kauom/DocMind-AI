-- =====================================================================
-- DocMind AI — Production Schema (PostgreSQL 16+ / Supabase)
-- =====================================================================

CREATE EXTENSION IF NOT EXISTS "pgcrypto";
CREATE EXTENSION IF NOT EXISTS "vector";
CREATE EXTENSION IF NOT EXISTS "pg_trgm";

-- =====================================================================
-- 1. PROFILES
-- =====================================================================
CREATE TABLE public.profiles (
    id                  UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
    full_name           TEXT,
    avatar_url          TEXT,
    role                TEXT NOT NULL DEFAULT 'member',
    subscription_plan   TEXT NOT NULL DEFAULT 'free',
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    CONSTRAINT chk_profiles_role CHECK (role IN ('member', 'admin', 'owner')),
    CONSTRAINT chk_profiles_subscription_plan CHECK (subscription_plan IN ('free', 'pro', 'team', 'enterprise'))
);
COMMENT ON TABLE public.profiles IS 'Extended profile data for each Supabase auth.users account.';
COMMENT ON COLUMN public.profiles.id IS 'Matches auth.users.id; profile is deleted when the auth user is deleted.';
COMMENT ON COLUMN public.profiles.role IS 'Application-level role: member, admin, or owner.';
COMMENT ON COLUMN public.profiles.subscription_plan IS 'Current billing plan tier.';

CREATE INDEX idx_profiles_subscription_plan ON public.profiles(subscription_plan);

ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

CREATE POLICY profiles_select_own ON public.profiles
    FOR SELECT USING (id = auth.uid());
CREATE POLICY profiles_update_own ON public.profiles
    FOR UPDATE USING (id = auth.uid());
CREATE POLICY profiles_insert_own ON public.profiles
    FOR INSERT WITH CHECK (id = auth.uid());

-- =====================================================================
-- 2. PROJECTS
-- =====================================================================
CREATE TABLE public.projects (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    owner_id            UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
    name                TEXT NOT NULL,
    description         TEXT,
    icon                TEXT,
    color               TEXT,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    deleted_at          TIMESTAMPTZ,
    CONSTRAINT chk_projects_name_not_blank CHECK (btrim(name) <> '')
);
COMMENT ON TABLE public.projects IS 'Workspaces grouping documents, chats, and reports for a single owner.';
COMMENT ON COLUMN public.projects.owner_id IS 'Profile that owns and administers this project.';
COMMENT ON COLUMN public.projects.deleted_at IS 'Soft-delete timestamp; NULL means the project is active.';

CREATE INDEX idx_projects_owner_id ON public.projects(owner_id);
CREATE INDEX idx_projects_deleted_at ON public.projects(deleted_at) WHERE deleted_at IS NULL;

ALTER TABLE public.projects ENABLE ROW LEVEL SECURITY;

CREATE POLICY projects_owner_all ON public.projects
    FOR ALL USING (owner_id = auth.uid()) WITH CHECK (owner_id = auth.uid());

-- =====================================================================
-- 3. DOCUMENTS
-- =====================================================================
CREATE TABLE public.documents (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id          UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
    original_name       TEXT NOT NULL,
    mime_type           TEXT NOT NULL,
    storage_path        TEXT NOT NULL,
    file_size           BIGINT NOT NULL CHECK (file_size >= 0),
    pages               INTEGER CHECK (pages IS NULL OR pages >= 0),
    language            TEXT,
    document_type       TEXT,
    extracted_text      TEXT,
    processing_stage    TEXT NOT NULL DEFAULT 'ingestion',
    upload_status       TEXT NOT NULL DEFAULT 'pending',
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    deleted_at          TIMESTAMPTZ,
    CONSTRAINT uq_documents_project_storage_path UNIQUE (project_id, storage_path),
    CONSTRAINT chk_documents_processing_stage CHECK (
        processing_stage IN ('ingestion', 'classification', 'analysis', 'review', 'completed', 'failed')
    ),
    CONSTRAINT chk_documents_upload_status CHECK (
        upload_status IN ('pending', 'uploading', 'processing', 'ready', 'failed')
    )
);
COMMENT ON TABLE public.documents IS 'Uploaded source documents belonging to a project.';
COMMENT ON COLUMN public.documents.storage_path IS 'Object key within the "documents" Supabase Storage bucket.';
COMMENT ON COLUMN public.documents.processing_stage IS 'Current stage in the AI ingestion/analysis pipeline.';
COMMENT ON COLUMN public.documents.upload_status IS 'Lifecycle status of the raw file upload.';
COMMENT ON COLUMN public.documents.deleted_at IS 'Soft-delete timestamp; NULL means the document is active.';

CREATE INDEX idx_documents_project_id ON public.documents(project_id);
CREATE INDEX idx_documents_document_type ON public.documents(document_type);
CREATE INDEX idx_documents_upload_status ON public.documents(upload_status);
CREATE INDEX idx_documents_deleted_at ON public.documents(deleted_at) WHERE deleted_at IS NULL;
CREATE INDEX idx_documents_extracted_text_trgm ON public.documents USING gin (extracted_text gin_trgm_ops);

ALTER TABLE public.documents ENABLE ROW LEVEL SECURITY;

CREATE POLICY documents_via_project_ownership ON public.documents
    FOR ALL USING (
        EXISTS (SELECT 1 FROM public.projects p WHERE p.id = project_id AND p.owner_id = auth.uid())
    )
    WITH CHECK (
        EXISTS (SELECT 1 FROM public.projects p WHERE p.id = project_id AND p.owner_id = auth.uid())
    );

-- =====================================================================
-- 4. DOCUMENT EMBEDDINGS (pgvector)
-- =====================================================================
CREATE TABLE public.document_embeddings (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id         UUID NOT NULL REFERENCES public.documents(id) ON DELETE CASCADE,
    chunk_index         INTEGER NOT NULL CHECK (chunk_index >= 0),
    chunk_text          TEXT NOT NULL,
    embedding           VECTOR(1536) NOT NULL,
    metadata            JSONB NOT NULL DEFAULT '{}'::jsonb,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    CONSTRAINT uq_document_embeddings_doc_chunk UNIQUE (document_id, chunk_index)
);
COMMENT ON TABLE public.document_embeddings IS 'Chunked text and vector embeddings supporting semantic search / RAG retrieval.';
COMMENT ON COLUMN public.document_embeddings.chunk_index IS 'Sequential position of this chunk within the source document.';
COMMENT ON COLUMN public.document_embeddings.embedding IS '1536-dimension vector embedding of chunk_text.';
COMMENT ON COLUMN public.document_embeddings.metadata IS 'Chunk metadata: page number, section heading, offsets, etc.';

CREATE INDEX idx_document_embeddings_document_id ON public.document_embeddings(document_id);
CREATE INDEX idx_document_embeddings_vector ON public.document_embeddings
    USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);

ALTER TABLE public.document_embeddings ENABLE ROW LEVEL SECURITY;

CREATE POLICY document_embeddings_via_project_ownership ON public.document_embeddings
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM public.documents d
            JOIN public.projects p ON p.id = d.project_id
            WHERE d.id = document_id AND p.owner_id = auth.uid()
        )
    );

-- =====================================================================
-- 5. PROMPTS
-- =====================================================================
CREATE TABLE public.prompts (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name                TEXT NOT NULL,
    version             INTEGER NOT NULL DEFAULT 1 CHECK (version >= 1),
    system_prompt       TEXT NOT NULL,
    description         TEXT,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    CONSTRAINT uq_prompts_name_version UNIQUE (name, version)
);
COMMENT ON TABLE public.prompts IS 'Versioned system prompt definitions used to drive AI job execution.';
COMMENT ON COLUMN public.prompts.name IS 'Logical prompt identifier, e.g. risk_detection, executive_summary.';
COMMENT ON COLUMN public.prompts.version IS 'Monotonically increasing version number per prompt name.';

CREATE INDEX idx_prompts_name ON public.prompts(name);

ALTER TABLE public.prompts ENABLE ROW LEVEL SECURITY;

CREATE POLICY prompts_read_all_authenticated ON public.prompts
    FOR SELECT USING (auth.role() = 'authenticated');

-- =====================================================================
-- 6. AI JOBS
-- =====================================================================
CREATE TABLE public.ai_jobs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id         UUID NOT NULL REFERENCES public.documents(id) ON DELETE CASCADE,
    prompt_id           UUID REFERENCES public.prompts(id) ON DELETE SET NULL,
    model_name          TEXT NOT NULL,
    job_type            TEXT NOT NULL,
    status              TEXT NOT NULL DEFAULT 'queued',
    tokens_used         INTEGER CHECK (tokens_used IS NULL OR tokens_used >= 0),
    estimated_cost      NUMERIC(10, 6) CHECK (estimated_cost IS NULL OR estimated_cost >= 0),
    processing_time_ms  INTEGER CHECK (processing_time_ms IS NULL OR processing_time_ms >= 0),
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    completed_at        TIMESTAMPTZ,
    CONSTRAINT chk_ai_jobs_status CHECK (
        status IN ('queued', 'running', 'completed', 'failed', 'cancelled')
    ),
    CONSTRAINT chk_ai_jobs_completed_after_created CHECK (
        completed_at IS NULL OR completed_at >= created_at
    )
);
COMMENT ON TABLE public.ai_jobs IS 'A unit of AI pipeline work executed against a document.';
COMMENT ON COLUMN public.ai_jobs.job_type IS 'Analysis type, e.g. summary, entity_extraction, risk_detection, classification.';
COMMENT ON COLUMN public.ai_jobs.model_name IS 'Identifier of the underlying model/provider used for this job.';
COMMENT ON COLUMN public.ai_jobs.tokens_used IS 'Total tokens consumed by this job, for cost tracking.';
COMMENT ON COLUMN public.ai_jobs.estimated_cost IS 'Estimated monetary cost of this job in USD.';

CREATE INDEX idx_ai_jobs_document_id ON public.ai_jobs(document_id);
CREATE INDEX idx_ai_jobs_prompt_id ON public.ai_jobs(prompt_id);
CREATE INDEX idx_ai_jobs_status ON public.ai_jobs(status);

ALTER TABLE public.ai_jobs ENABLE ROW LEVEL SECURITY;

CREATE POLICY ai_jobs_via_project_ownership ON public.ai_jobs
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM public.documents d
            JOIN public.projects p ON p.id = d.project_id
            WHERE d.id = document_id AND p.owner_id = auth.uid()
        )
    );

-- =====================================================================
-- 7. AI OUTPUTS
-- =====================================================================
CREATE TABLE public.ai_outputs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    job_id              UUID NOT NULL REFERENCES public.ai_jobs(id) ON DELETE CASCADE,
    output_type         TEXT NOT NULL,
    content             JSONB NOT NULL DEFAULT '{}'::jsonb,
    confidence_score    NUMERIC(4, 3) CHECK (confidence_score IS NULL OR (confidence_score >= 0 AND confidence_score <= 1)),
    review_status       TEXT NOT NULL DEFAULT 'unreviewed',
    created_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
    CONSTRAINT chk_ai_outputs_review_status CHECK (
        review_status IN ('unreviewed', 'verified', 'flagged', 'rejected')
    )
);
COMMENT ON TABLE public.ai_outputs IS 'Structured result produced by an AI job, stored as flexible JSON.';
COMMENT ON COLUMN public.ai_outputs.output_type IS 'Type of output payload, e.g. executive_summary, entities, risks, action_items.';
COMMENT ON COLUMN public.ai_outputs.confidence_score IS 'Model or reviewer confidence score between 0 and 1.';
COMMENT ON COLUMN public.ai_outputs.review_status IS 'Fact-consistency review outcome for this output.';

CREATE INDEX idx_ai_outputs_job_id ON public.ai_outputs(job_id);
CREATE INDEX idx_ai_outputs_output_type ON public.ai_outputs(output_type);
CREATE INDEX idx_ai_outputs_content_gin ON public.ai_outputs USING gin (content);

ALTER TABLE public.ai_outputs ENABLE ROW LEVEL SECURITY;

CREATE POLICY ai_outputs_via_project_ownership ON public.ai_outputs
    FOR ALL USING (
        EXISTS (
            SELECT 1 FROM public.ai_jobs j
            JOIN public.documents d ON d.id = j.document_id
            JOIN public.projects p ON p.id = d.project_id
            WHERE j.id = job_id AND p.owner_id = auth.uid()
        )
    );

-- =====================================================================
-- UPDATED_AT MAINTENANCE TRIGGER
-- =====================================================================
CREATE OR REPLACE FUNCTION public.set_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = now();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;
COMMENT ON FUNCTION public.set_updated_at() IS 'Trigger function that stamps updated_at with the current timestamp on row update.';

CREATE TRIGGER trg_profiles_updated_at BEFORE UPDATE ON public.profiles
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_projects_updated_at BEFORE UPDATE ON public.projects
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_documents_updated_at BEFORE UPDATE ON public.documents
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_chats_updated_at BEFORE UPDATE ON public.chats
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_reports_updated_at BEFORE UPDATE ON public.reports
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
CREATE TRIGGER trg_settings_updated_at BEFORE UPDATE ON public.settings
    FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();
